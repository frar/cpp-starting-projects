102  t1 = new Task_T1("Default message.");
103  FixedIntervalPeriodicity periodicity_t1 = {0, 0, 0, 0, 0, 10};
104  t1->addFixedIntervalPeriodicity(periodicity_t1);
105  Task_T2 *t2 = new Task_T2();
106  FixedIntervalPeriodicity periodicity_t2 = {0, 0, 0, 0, 0, 30};
107  t2->addFixedIntervalPeriodicity(periodicity_t2);
108 
109  // Run Scheduler Timer

111  mSchedulerTimer->addTask(t1);
112  mSchedulerTimer->addTask(t2);
113

115  connect(mSchedulerTimer, SIGNAL(timeout()), mSchedulerTimer, SLOT(executeNextTask()));
